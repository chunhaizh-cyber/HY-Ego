    START["旧鱼巢自我线程逻辑提取开始"]
    SRC["只读旧项目代码与旧流程图"]
    INIT_CALL["自我线程类::自我初始化"]
    INIT_FAST{"已经初始化且非故障"}
    INIT_RETURN["逻辑内返回：初始化快路径成功"]
    INIT_ENV["执行自我初始化并记录步骤耗时"]
    INIT_INSTINCT["注册本能函数执行闭环"]
    INIT_DELAY["默认本能方法专属规格延后确认"]
    INIT_DEMAND["刷新需求列表并确保需求相关任务"]
    INIT_MAIN["选择当前主任务候选并同步当前主需求"]
    INIT_MARK["刷新初始化标记与最近初始化结果"]
    INIT_OK{"初始化完成"}
    INIT_FAIL["追根因解决：初始化异常或标记不及预期进入故障收口"]

    RUN_ENTER["线程函数进入运行态"]
    LOOP["while 未收到停止请求"]
    ROUND["执行主循环一轮"]
    FREEZE["冻结本轮治理消息批次"]
    STOP1{"停止请求"}
    STOP_RETURN["逻辑内返回：停止请求中"]
    NORMALIZE["规范化消息并落一次特征账"]
    PREEMPT1["抢占处理：成功结果、派生需求、观察事实承接、执行前许可"]
    DERIVED1{"存在待落账派生需求"}
    WAIT_DERIVED["逻辑内返回：暂停本轮任务派发，等待派生需求入树"]
    SECONDARY["重算受影响二次特征"]
    FRAME["刷新线程上下文并生成治理帧"]
    DEMAND["整理需求并做根层重判"]
    DISPATCH["生成并执行主派发决议"]
    TASK_REQ["向任务管理界面提交任务治理请求或调度意图"]
    PREEMPT2["主派发后再次抢占落账与许可审批"]
    DERIVED2{"主派发后出现待落账派生需求"}
    WAIT_DERIVED2["逻辑内返回：暂停任务巡检，下一轮优先冻结治理邮箱"]
    TASK_UP["提取并上报任务管理后台上行消息"]
    SERVICE["执行服务治理周期维护"]
    SELF_CHECK["休眠期自检报告修复门控"]
    ROUND_OK["更新 Tick、最近时间、首轮完成和最近循环结果"]
    WAIT_WAKE["等待 Tick 间隔或治理邮箱唤醒"]

    MSG_IN["外部反馈、自检报告、任务管理上行消息"]
    POST["投递治理消息"]
    POST_STOP{"停止请求中"}
    POST_SKIP["逻辑内返回：拒绝入队"]
    POST_FILL["补齐产生时间、来源链、优先级、可抢占、消息主键和载荷键"]
    POST_MERGE{"幂等键已存在"}
    POST_UPDATE["合并既有事件的优先级、摘要和附加载荷"]
    POST_APPEND["追加到治理 mailbox 并唤醒线程"]

    PROTOCOL["结构_治理消息"]
    PAYLOAD["附加载荷：线程状态、派生需求、结果状态、观察承接、执行前许可、自检报告、控制面板刷新请求"]
    BATCH["消息批次执行器拆分抢占消息与普通消息"]
    ACTION["消息处理器计算动作：需求队列刷新、根层重判、主派发、二次特征、结算候选"]

    TASK_UI["任务管理界面线程"]
    TASK_WORKER["任务管理工作线程"]
    TASK_RESULT["任务工作结果与结构_任务管理上行消息"]
    SELF_MAILBOX["转换为自我线程结构_治理消息"]
    PERMIT["执行前许可请求由自我线程审批并回传许可结果"]

    EXCEPTION["运行期异常"]
    FAULT["追根因解决：故障收口，生命周期置故障，停止请求生效"]
    EXIT["退出循环并切换已停止"]

    RULE_THREAD["边界：线程不是动作来源"]
    RULE_TASK["边界：自我线程不得直接写任务状态"]
    RULE_TEXT["边界：日志、摘要、控制台和显示不承载机器事实"]
    RULE_MIGRATE["边界：本图只生成流程图依据，不迁移旧函数体"]
    RULE_EXCLUDE["排除：SQL、控制面板实现、D455、体素、外设接入不借本图进入实施"]

    DERIVED2 -->|否| TASK_UP --> SERVICE --> SELF_CHECK --> ROUND_OK --> WAIT_WAKE --> LOOP

    POST_FILL --> PROTOCOL --> PAYLOAD

    SRC -.->|排除项| RULE_EXCLUDE
