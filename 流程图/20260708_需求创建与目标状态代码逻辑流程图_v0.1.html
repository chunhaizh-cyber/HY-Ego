需求创建与目标状态代码逻辑流程图

更新时间：2026-07-08

本图是第 5 项“需求创建与目标状态流程”的代码逻辑流程图，承接第 4 项“特征与状态材料”输出的当前特征状态材料和抽象目标状态材料。第一轮简化入口只作为兼容 / 验证入口，不提供完整任务承接材料。本图不生成代码实施许可。

flowchart TD
    A["上游请求 / 语素材料 / 特征状态材料"] --> B["降权为需求创建请求"]
    B --> C{"请求类型"}

    C -- "只创建需求身份" --> N1["需求服务::创建需求"]
    N1 --> N2["主信息仓库创建主信息"]
    N2 --> N3["节点仓库创建 需求 节点"]
    N3 --> RVN["读回验证：需求节点"]

    C -- "创建目标状态需求(兼容 / 验证入口)" --> D1["需求服务::创建目标状态需求(目标宿主, 目标特征, 当前特征状态材料, 目标状态值)"]
    D1 --> D2{"目标宿主有效 && 目标特征有效"}
    D2 -- "否" --> RJ["拒绝：不创建需求 / 结构不变化"]
    D2 -- "是" --> D3["特征服务读取当前 I64 特征状态材料"]
    D3 --> D4{"当前状态材料可读"}
    D4 -- "否" --> RJ
    D4 -- "是" --> D5["状态服务::创建抽象状态(目标状态值)"]
    D5 --> D6{"目标状态节点有效"}
    D6 -- "否" --> FS["失败收口：不返回有效需求 / 不允许半结构作为事实"]
    D6 -- "是" --> D7["需求服务::创建需求"]
    D7 --> D8["关系仓库写 需求->目标存在 引用关系"]
    D8 --> D9["关系仓库写 需求->目标特征 引用关系"]
    D9 --> D10["关系仓库写 需求->当前特征状态材料 引用关系"]
    D10 --> D11["关系仓库写 需求->目标状态 模板关系"]
    D11 --> RVD["读回验证：需求节点 / 目标状态 / 目标宿主 / 目标特征 / 当前特征状态材料 / 模板关系"]

    C -- "创建目标状态需求(承接材料完整入口)" --> E1["需求服务::创建目标状态需求(主体, 目标宿主, 场景, 目标特征, 当前特征状态材料, 目标状态值)"]
    E1 --> E2{"主体有效 && 目标宿主有效 && 场景有效 && 目标特征有效 && 主体 / 目标宿主角色已明确"}
    E2 -- "否" --> RJ
    E2 -- "是" --> E3["特征服务读取当前 I64 特征状态材料"]
    E3 --> E4{"当前状态材料可读"}
    E4 -- "否" --> RJ
    E4 -- "是" --> E5["状态服务::创建抽象状态(目标状态值)"]
    E5 --> E6{"目标状态节点有效"}
    E6 -- "否" --> FS
    E6 -- "是" --> E7["需求服务::创建需求"]
    E7 --> E8["关系仓库写 需求->主体 引用关系(主体顺序号)"]
    E8 --> E9["关系仓库写 需求->目标宿主 引用关系(目标宿主顺序号)"]
    E9 --> E10["关系仓库写 需求->场景 引用关系(场景顺序号)"]
    E10 --> E11["关系仓库写 需求->目标特征 引用关系(目标特征顺序号)"]
    E11 --> E12["关系仓库写 需求->当前特征状态材料 引用关系(当前材料顺序号)"]
    E12 --> E13["关系仓库写 需求->目标状态 模板关系"]
    E13 --> RVE["读回验证：需求节点 / 目标状态 / 主体 / 目标宿主 / 场景 / 目标特征 / 当前特征状态材料 / 模板关系"]

    C -- "读取需求目标状态" --> R1["需求服务::读取需求目标状态"]
    R1 --> R2{"需求有效 && 唯一模板目标为 状态 节点"}
    R2 -- "否" --> RJ
    R2 -- "是" --> R3["返回目标状态节点"]
    R3 --> READ["只读方：任务服务 / 需求结算 / 显示层只读"]

    C -- "读取需求承接材料" --> M1["需求服务::读取需求承接材料"]
    M1 --> M2["读取需求主体 / 目标宿主 / 场景 / 目标状态"]
    M2 --> M3{"四项材料均唯一且类型正确"}
    M3 -- "否" --> RJ
    M3 -- "是" --> M4["输出需求承接材料"]
    M4 --> T1["任务服务::按需求创建任务 只读承接材料"]

    C -- "目标状态满足预判" --> J1["需求服务::目标状态满足预判 / 结算预览"]
    J1 --> J2{"需求引用目标特征和当前特征状态材料"}
    J2 -- "否" --> JF["返回 证据不足 / 不写事实"]
    J2 -- "是" --> J3["读取需求目标状态"]
    J3 --> J4["状态服务读取目标状态值"]
    J4 --> J5["特征服务读取当前 I64 特征状态材料"]
    J5 --> J6{"目标状态值和当前材料均可读"}
    J6 -- "否" --> JF
    J6 -- "是" --> J7{"当前材料 >= 目标状态值"}
    J7 -- "是" --> J8["返回 已满足"]
    J7 -- "否" --> J9["返回 未满足"]
    J8 --> NR["局部枚举结果 / 预览材料，不写持久化结算记录"]
    J9 --> NR

    C -- "旧需求树 / SQL / 显示材料" --> X1["排除或后续专项"]
    X1 --> X2["父子需求 / 逻辑组织 / 方向签名 / 枚举目标 转第 6 项及后续"]
    X1 --> X3["SQL / 控制面板 / 显示镜像 不参与机器事实"]
    X1 --> X4["旧需求主信息字段不得直接迁移"]

    RVN --> RV1{"读回符合预期"}
    RVD --> RV1
    RVE --> RV1
    RV1 -- "否" --> LE["逻辑错误：登记断点 / 停止后续施工计划推进"]
    RV1 -- "是" --> OUT["输出需求节点句柄和目标状态结构"]
    OUT --> READ

    N2 -. "主信息创建失败" .-> NFS["身份创建失败收口：不返回有效需求 / 需求节点不可读"]
    N3 -. "需求节点创建失败" .-> NFS
    NFS --> RJ
    D5 -. "目标状态创建后后续失败" .-> FS
    D7 -. "创建失败" .-> FS
    D8 -. "关系失败" .-> FS
    D9 -. "关系失败" .-> FS
    D10 -. "关系失败" .-> FS
    D11 -. "关系失败" .-> FS
    E7 -. "创建失败" .-> FS
    E8 -. "关系失败" .-> FS
    E9 -. "关系失败" .-> FS
    E10 -. "关系失败" .-> FS
    E11 -. "关系失败" .-> FS
    E12 -. "关系失败" .-> FS
    E13 -. "关系失败" .-> FS
    FS -. "覆盖本次创建的目标状态 / 需求节点 / 关系" .-> CLEAN["失败后不得留下可读半结构"]
    CLEAN --> RJ
