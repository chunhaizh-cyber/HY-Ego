特征体系服务分层迁移代码逻辑流程图 v0.1

更新时间：2026-07-14。表达 #267 在 #274 正式接口复核通过后的隔离新分层路径。

flowchart TD
    A["调用方提交无令牌特征或二次特征业务请求"] --> B["特征业务服务 / 二次特征业务服务 / 特征状态组合器"]
    B --> C["入口参数检测：非零主键、句柄、原始类型、字段互斥和组成项"]
    C --> D{"入口是否符合业务前置条件"}
    D -->|"否"| L1["逻辑内返回：入口拒绝，结构不变化"]
    D -->|"是"| E{"请求类型"}

    E -->|"创建特征定义"| F1["特征定义主键"]
    E -->|"创建实例特征槽位"| F2["槽位主键 + 宿主(存在 / 场景 / 需求 / 任务 / 方法) + 特征定义"]
    E -->|"发布初始特征状态"| F3["特征值主键 + 已有槽位 + I64 / VecI64 / VecU64"]
    E -->|"创建槽位并发布初始状态"| F4["互异槽位 / 特征值主键 + 白名单宿主 + 定义 + 原始材料"]
    E -->|"创建组合二次特征"| F5["二次特征主键 + 有序长期组成项组"]
    E -->|"读取宿主特征状态"| F6["白名单宿主 + 特征定义"]

    F1 --> G["数据操作层取得共享许可并按索引候选读取权威结构"]
    F2 --> G
    F3 --> G
    F4 --> G
    F5 --> G
    F6 --> R0["一次共享结构许可内读取宿主、槽位、模板、当前值和完整关系句柄"]

    G --> H{"同一事实是否已经存在"}
    H -->|"完全一致"| L2["逻辑内返回：幂等读回"]
    H -->|"主键或结构冲突"| L1
    H -->|"不存在"| I["数据操作.特征体系提交类型化写入规格"]
    I --> J["结构写入执行器取得一次独占许可并构造不透明会话"]
    J --> K["许可内重读宿主、定义、槽位、组成项、主键和关系当前性"]
    K --> M{"当前材料是否仍符合请求"}
    M -->|"否"| L3["逻辑内返回：版本漂移；第一笔写入前结构不变化"]
    M -->|"是"| N{"结构写入路径"}

    N -->|"特征定义"| P1["主信息候选 -> 特征节点候选 -> 主键"]
    N -->|"实例槽位"| P2["主信息候选 -> 特征槽位节点候选 -> 主键 -> 宿主归属 -> 定义模板"]
    N -->|"初始状态"| P3["复核槽位无当前值 -> 主信息候选 -> 特征值节点候选 -> 主键 -> 槽位归属"]
    N -->|"槽位加初始状态"| P4["同会话创建槽位结构和唯一特征值结构"]
    N -->|"组合二次特征"| P5["主信息候选 -> 二次特征节点候选 -> 主键 -> 按 0..n-1 写引用关系"]

    P1 --> Q["同令牌读回主信息、节点、主键和完整关系"]
    P2 --> Q
    P5 --> Q
    P3 --> V["#274 参与者登记初始 I64 版本或 Vec 完整值，版本固定为 1"]
    P4 --> V
    V --> Q
    Q --> S{"结构和业务规格是否完全一致"}
    S -->|"否"| R1["追根因：内部不一致，禁止提交"]
    S -->|"是"| T["请求提交；参与者准备；会话确认；参与者完成提交"]
    T --> U["释放许可并返回值式业务结果"]

    J -->|"许可失败、侧表锁竞争、异常或无提交决定"| RB["参与者和会话在许可释放前精确撤销"]
    R1 --> RB
    RB --> RC["释放许可后权威重读四仓库和特征值侧表"]
    RC --> RD{"是否为并发形成的同一完整事实"}
    RD -->|"是"| L2
    RD -->|"否且全部回前态"| L4["返回许可失败、版本漂移或结构化失败"]
    RD -->|"否且存在残留"| R2["追根因：停止后继计划并登记断点"]

    R0 --> R3["#274 侧表访问器在同一结构许可内 try_lock_shared，不访问仓库"]
    R3 --> R4{"侧表锁是否取得"}
    R4 -->|"否"| L5["逻辑内返回：许可竞争，不拼接混合快照"]
    R4 -->|"是"| R5["组合主信息 I64 或 Vec 副本、类型和版本"]
    R5 --> R6{"槽位、定义、唯一当前值和原始材料是否完整"}
    R6 -->|"无槽位或无当前值"| L6["逻辑内返回：未找到"]
    R6 -->|"完整"| R7["返回宿主特征状态值式材料"]
    R6 -->|"重复、矛盾或半结构"| R8["追根因：不选择第一个，不生成后续产物"]
