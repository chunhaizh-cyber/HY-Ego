```powershell
$期望 = @('海中鱼巣/适配/处理.D455变化跟踪.ixx','海中鱼巣/适配/自检.D455变化跟踪.ixx','实施记录/20260724_D455-P5_PER-C5变化跟踪产品代码实施_Codex断点清单.md') | Sort-Object
$实际 = @(git -c core.quotepath=false diff --name-only "$计划段起点..HEAD") | Sort-Object
if (Compare-Object $期望 $实际) { exit 1 }
if ((@(rg -n "D455变化跟踪ABI版本\s*=\s*2" '海中鱼巣/适配/处理.D455变化跟踪.ixx')).Count -ne 1) { exit 1 }
if ((@(rg -n "形成D455稳定观察产品|形成D455扫描变化产品|形成D455目标跟踪产品" '海中鱼巣/适配/处理.D455变化跟踪.ixx')).Count -lt 3) { exit 1 }
```
